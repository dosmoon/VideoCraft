# ── Embedded AI (in-process, no Docker) ─────────────────────────────────────
# See docs/draft/tech-selection-embedded-ai.md.
#
# Two install profiles:
#   CPU only    — what `pip install -r requirements.txt` gets you below.
#   CUDA 12 GPU — extra commands listed below; required for usable speed
#                 on Whisper-turbo and Qwen3-8B.
#
# CPU profile (default):
llama-cpp-python==0.3.22  # see CPU wheel install note below
faster-whisper==1.2.1     # CTranslate2-backed Whisper; works on CPU and CUDA
                           # without a separate wheel (CT2 picks at runtime).
                           # Bundles silero VAD via vad_filter=True.
#
# DO NOT install `torch` into this venv. ctranslate2.converters.transformers
# has `try: import torch`; if torch is present it eagerly loads its bundled
# cudnn DLLs and clashes with our nvidia-cudnn-cu12 on PATH (Error 127).
# faster-whisper itself doesn't need torch — converters are offline-only.
#
# llama-cpp-python wheels are published outside PyPI — building from source
# on Windows breaks (vendored llama.cpp paths exceed MAX_PATH). Install via:
#   pip install llama-cpp-python==0.3.22 \
#     --extra-index-url https://abetlen.github.io/llama-cpp-python/whl/cpu \
#     --only-binary=:all:
#
# ── GPU profile (CUDA 12 / cuDNN 9) ────────────────────────────────────────
# Adds CUDA wheels for llama-cpp-python + the nvidia-* runtime DLLs that
# faster-whisper / CTranslate2 also pick up automatically. Run AFTER the
# CPU install (or skip the CPU install entirely):
#
#   pip install --force-reinstall --no-deps llama-cpp-python==0.3.22 \
#     --extra-index-url https://abetlen.github.io/llama-cpp-python/whl/cu124 \
#     --only-binary=:all:
#
#   pip install --no-deps \
#     nvidia-cuda-runtime-cu12 nvidia-cublas-cu12 nvidia-cudnn-cu12 nvidia-cufft-cu12
#
# Activation is automatic: core/gpu.py prepends the nvidia/* DLL dirs to
# PATH at provider import, and providers default to "auto" which picks
# GPU when these wheels + an NVIDIA driver are detected.
